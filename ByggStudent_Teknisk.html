-- SQL for å lage tabellene (lim inn i Supabase)
CREATE TABLE brukere (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  navn text NOT NULL,
  epost text UNIQUE NOT NULL,
  plan text DEFAULT 'basis',
  aktiv boolean DEFAULT false,
  opprettet timestamp DEFAULT now()
);

CREATE TABLE bookinger (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  bruker_id uuid REFERENCES brukere(id),
  dato timestamp,
  fag text,
  melding text,
  status text DEFAULT 'venter',
  betalt boolean DEFAULT false,
  opprettet timestamp DEFAULT now()
);

CREATE TABLE pdf_dokumenter (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  tittel text NOT NULL,
  drive_url text,
  kategori text,
  krever_plan text DEFAULT 'gratis',
  aktiv boolean DEFAULT true
);

// Registrer ny student
async function registrer(navn, epost, passord, plan) {
  // 1. Opprett bruker i Auth
  const { data, error } = await supabase.auth
    .signUp({ email: epost, password: passord });

  // 2. Lagre i brukere-tabellen
  await supabase.from('brukere').insert({
    navn, epost, plan,
    aktiv: false // Du aktiverer manuelt
  });

  // 3. Send e-post til deg (EmailJS)
  sendAdminVarsel(navn, epost, plan);
}

// Logg inn
async function loggInn(epost, passord) {
  const { data } = await supabase.auth
    .signInWithPassword({ email, password });
  if (data.user) lastDashboard(data.user);
}

// Sjekk tilgang til PDF
async function sjekkTilgang(bruker_id, pdf_id) {
  const bruker = await supabase
    .from('brukere')
    .select('plan,aktiv')
    .eq('id', bruker_id)
    .single();

  const pdf = await supabase
    .from('pdf_dokumenter')
    .select('krever_plan,drive_url')
    .eq('id', pdf_id)
    .single();

  const planer = ['gratis','basis','student','pro'];
  const harTilgang = bruker.aktiv &&
    planer.indexOf(bruker.plan) >=
    planer.indexOf(pdf.krever_plan);

  return harTilgang ? pdf.drive_url : null;
}

// api/paypal-webhook.js (Vercel serverless funksjon)
// Fil: /api/paypal-webhook.js

export default async function handler(req, res) {
  const event = req.body;

  // Sjekk at det er en betaling
  if (event.event_type === 'PAYMENT.CAPTURE.COMPLETED') {
    const epost = event.resource.payer.email_address;
    const beloep = event.resource.amount.value;

    // Finn riktig plan basert på beløp
    const plan = beloep >= 379 ? 'pro'
                : beloep >= 179 ? 'student' : 'basis';

    // Aktiver i Supabase
    const supabase = createClient(
      process.env.SUPABASE_URL,
      process.env.SUPABASE_SERVICE_KEY // Service key!
    );

    await supabase
      .from('brukere')
      .update({ aktiv: true, plan })
      .eq('epost', epost);

    // Send velkomst-e-post via EmailJS
    await sendVelkomstEpost(epost, plan);
  }

  res.json({ status: 'ok' });
}

// Hent PDF-liste for innlogget student
async function hentPDFer(bruker) {
  // Hent brukerens plan
  const { data: b } = await supabase
    .from('brukere')
    .select('plan')
    .eq('id', bruker.id)
    .single();

  // Hent alle PDF-er
  const { data: pdfer } = await supabase
    .from('pdf_dokumenter')
    .select('*')
    .eq('aktiv', true);

  const planer = ['gratis','basis','student','pro'];
  const brukerNivaa = planer.indexOf(b.plan);

  // Marker hvilke brukeren har tilgang til
  return pdfer.map(pdf => ({
    ...pdf,
    harTilgang: brukerNivaa >=
      planer.indexOf(pdf.krever_plan),
    // Skjul URL hvis ikke tilgang
    drive_url: brukerNivaa >=
      planer.indexOf(pdf.krever_plan)
      ? pdf.drive_url : null
  }));
}

// Åpne PDF i viewer
function aapnePDF(pdf) {
  if (!pdf.harTilgang) {
    visOppgraderModal(); // Vis "Oppgrader"
    return;
  }
  // Åpne i Google Docs viewer
  const viewer = `https://docs.google.com/
    viewer?url=${pdf.drive_url}&embedded=true`;
  document.getElementById('pdf-iframe')
    .src = viewer;
}
